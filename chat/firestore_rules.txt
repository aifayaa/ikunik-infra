// Firestore access rules for the chat feature
// Last updated 2025-06-16

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
  	allow read, write: if request.auth != null && request.time < timestamp.date(2025, 6, 30);
    
    match /channels/{channelId} {
      function getChannel() {
      	let channel = get(/databases/$(database)/documents/channels/$(channelId)).data;
        return channel;
      }
      
      function isValidInitialChannelData() {
        let fields = ['name', 'description', 'ownerId', 'createdAt', 'isPublic', 'isDeleted', 'members', 'bannedUsers'];
        let data = request.resource.data;
        return data.keys().hasAll(fields) &&
               data.keys().size() == fields.size() && // Ensure no extra fields
               data.name is string &&
               data.name.size() > 0 && // Non-empty string
               data.description is string &&
               data.ownerId is string &&
               data.ownerId.size() > 0 && // Non-empty string
               data.ownerId == request.auth.uid && // Must be the authenticated user
               data.createdAt is timestamp &&
               data.isPublic is bool &&
               data.isDeleted is bool &&
               data.isDeleted == false && // Must be false at creation
               data.members.size() == 1 &&
               data.members[0] == request.auth.uid &&
               data.bannedUsers.size() == 0;
      }

      function isValidUpdatedChannelData() {
        let updatedKeys = request.resource.data.diff(resource.data).affectedKeys();
      	let data = request.resource.data;
      	return
        			resource.data.ownerId == request.auth.uid &&
              updatedKeys.hasOnly(['name', 'description', 'isPublic', 'isDeleted', 'members', 'bannedUsers']) &&
               data.get('name', '') is string &&
               data.get('name', 'x').size() > 0 &&
               data.get('description', '') is string &&
               data.get('isPublic', true) is bool &&
               data.get('isDeleted', true) is bool &&
               data.get('members', [request.auth.uid]).hasAll([request.auth.uid]) &&
               !(data.get('bannedUsers', []).hasAny([request.auth.uid]));
      }
      
      allow read: if
                    request.auth != null &&
                    resource.data.isDeleted == false &&
                    (resource.data.isPublic || request.auth.uid in resource.data.members);
      
      allow create: if request.auth != null && isValidInitialChannelData();
      
      allow update: if request.auth != null && isValidUpdatedChannelData();
      
      match /messages/{messageId} {
      	function allowedMessageUpdate() {
        	let updatedKeys = request.resource.data.diff(resource.data).affectedKeys();
        	return (
          	!updatedKeys.hasOnly(['reactions'])
          );
        }
        
        function isMemberOrPublic() {
          let channel = getChannel();
          return (channel.isPublic || request.auth.uid in channel.members) && !(request.auth.uid in channel.bannedUsers);
        }

    
        allow read: if
        							request.auth != null &&
                      isMemberOrPublic();
        
        allow create: if
                      	request.auth != null && 
                       	isMemberOrPublic() && 
                       	(
                       		request.resource.data.userId == request.auth.uid || (
                          	request.resource.data.type == 'system' &&
                            getChannel().ownerId == request.auth.uid
                          )
                        );
                        
        allow update: if 
        								request.auth != null &&
                        isMemberOrPublic() &&
                        (
                        	resource.data.userId == request.auth.uid ||
                        	allowedMessageUpdate()
                        )
      }
    }
    
    match /users/{userId} {
    	// Only allow getting items by ID
      allow get: if request.auth != null;
      allow list: if false;
      allow write: if false;
    }

    match /invitations/{invitationId} {
      function isEmitterOrReceiver() {
      	return
      				(
              	'fromUser' in resource.data &&
          	  	'id' in resource.data.fromUser &&
            		request.auth.uid == resource.data.fromUser.id
              ) || (
      					'toUser' in resource.data &&
          	  	'id' in resource.data.toUser &&
            		request.auth.uid == resource.data.toUser.id
              );
      }
      allow read: if request.auth != null && isEmitterOrReceiver();
      allow write: if false;
    }
  }
}
